
### 🌐 第四步：API和网络配置

#### 1. 创建API服务 `src/lib/api.ts`

```javascript
import axios from 'axios';

// 配置axios默认设置
axios.defaults.baseURL = 'http://localhost:8080';

// 请求拦截器
axios.interceptors.request.use(
  (config) => {
    // 可以在这里添加认证token
    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器
axios.interceptors.response.use(
  (response) => {
    return response;
  },
  (error) => {
    console.error('API Error:', error);
    return Promise.reject(error);
  }
);

// API函数定义
export const dubboApi = {
  // 获取服务列表
  getServices: async (port: number): Promise<string[]> => {
    const response = await axios.post('/api/dubbo/services', { port });
    return response.data.services || [];
  },
  
  // 获取方法列表
  getMethods: async (port: number, service: string): Promise<string[]> => {
    const response = await axios.post('/api/dubbo/methods', { port, service });
    return response.data.methods || [];
  },
  
  // 调用Dubbo方法
  invokeMethod: async (params: {
    port: number;
    serviceName: string;
    methodName: string;
    jsonParams: string;
  }) => {
    const response = await axios.post('/api/dubbo/invoke', params);
    return response.data;
  }
};

export default axios;
```

### 🪝 第六步：自定义Hooks开发

#### 1. 核心业务逻辑 `src/hooks/useDubboTester.ts`

```javascript
import { useState, useEffect } from 'react';
import { dubboApi } from '@/lib/api';
import { Port } from '@/lib/types';

// 默认端口配置
const getDefaultPorts = (): Port[] => [
  { id: 'port-1', name: '开发环境', host: 'localhost', port: 20880 },
  { id: 'port-2', name: '测试环境', host: 'localhost', port: 20881 },
  { id: 'port-3', name: '生产环境', host: 'localhost', port: 20882 },
];

export function useDubboTester() {
  // 端口状态
  const [ports, setPorts] = useState<Port[]>(getDefaultPorts());
  const [activePort, setActivePort] = useState<string>(ports[0]?.id || '');
  
  // 服务和方法状态
  const [services, setServices] = useState<string[]>([]);
  const [activeService, setActiveService] = useState<string | null>(null);
  const [methods, setMethods] = useState<string[]>([]);
  const [activeMethod, setActiveMethod] = useState<string | null>(null);
  
  // 请求和响应状态
  const [requestParams, setRequestParams] = useState<string>(
    JSON.stringify({}, null, 2)
  );
  const [responseData, setResponseData] = useState<any>(null);
  const [isLoading, setIsLoading] = useState(false);
  
  // 加载服务列表
  const loadServices = async () => {
    if (!activePort) return;
    
    const currentPort = ports.find(p => p.id === activePort);
    if (!currentPort) return;
    
    setIsLoading(true);
    try {
      const servicesList = await dubboApi.getServices(currentPort.port);
      setServices(servicesList);
    } catch (error) {
      console.error('加载服务列表失败:', error);
    } finally {
      setIsLoading(false);
    }
  };
  
  // 加载方法列表
  const loadMethods = async () => {
    if (!activePort || !activeService) return;
    
    const currentPort = ports.find(p => p.id === activePort);
    if (!currentPort) return;
    
    try {
      const methodsList = await dubboApi.getMethods(currentPort.port, activeService);
      setMethods(methodsList);
    } catch (error) {
      console.error('加载方法列表失败:', error);
    }
  };
  
  // 发送请求
  const handleSendRequest = async () => {
    if (!activePort || !activeService || !activeMethod) return;
    
    const currentPort = ports.find(p => p.id === activePort);
    if (!currentPort) return;
    
    setIsLoading(true);
    try {
      const result = await dubboApi.invokeMethod({
        port: currentPort.port,
        serviceName: activeService,
        methodName: activeMethod,
        jsonParams: requestParams
      });
      setResponseData(result);
    } catch (error) {
      setResponseData({ error: error.message });
    } finally {
      setIsLoading(false);
    }
  };
  
  // 当端口改变时重新加载服务
  useEffect(() => {
    loadServices();
  }, [activePort]);
  
  // 当服务改变时重新加载方法
  useEffect(() => {
    loadMethods();
  }, [activeService]);
  
  return {
    // 状态
    ports,
    activePort,
    setActivePort,
    services,
    activeService,
    setActiveService,
    methods,
    activeMethod,
    setActiveMethod,
    requestParams,
    setRequestParams,
    responseData,
    isLoading,
    
    // 方法
    loadServices,
    loadMethods,
    handleSendRequest,
  };
}
```

``` javascript
  requestParams, 
  setRequestParams, 
  className = '' 
}: RequestPanelProps) {

  const formatJson = () => {
    try {
      const formatted = JSON.stringify(JSON.parse(requestParams), null, 2);

      // 如果不是有效的JSON，忽略格式化
    }

  return (

      <div className="flex items-center justify-between mb-3">
        <h2 className="text-lg font-semibold text-gray-800">请求参数</h2>
```
